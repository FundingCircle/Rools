165     def fact( obj )
166       begin
167         # check if facts already exist for that class
168         # if so, we need to add it to the existing list
169         cls = obj.class.to_s.downcase
170         
171         if @facts.key? cls
172           logger.debug( "adding to facts: #{cls}") if logger
173           @facts[cls].fact_value << obj
174         else
175           logger.debug( "creating facts: #{cls}") if logger
176           arr = Array.new
177           arr << obj
178           proc = Proc.new { arr }
179           @facts[cls] = Facts.new(self, cls, proc ) 
180         end
181       rescue Exception=> e
182         logger.error e if logger
183       end
