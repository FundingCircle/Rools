assert(obj)

Used to create a working-set of rules for an object, and evaluate it against them. Returns a status, simply PASS or FAIL

 72:     def assert(obj)
 73:       @status = PASS
 74:       @assert = true
 75:       
 76:       # create a working-set of all parameter-matching, non-dependent rules
 77:       available_rules = @rules.values.select { |rule| rule.parameters_match?(obj) }
 78:       
 79:       begin
 80:         
 81:         # loop through the available_rules, evaluating each one,
 82:         # until there are no more matching rules available
 83:         begin # loop
 84:           
 85:           # the loop condition is reset to break by default after every iteration
 86:           matches = false
 87:           
 88:           available_rules.each do |rule|
 89:             # RuleCheckErrors are caught and swallowed and the rule that
 90:             # raised the error is removed from the working-set.
 91:             begin
 92:               if rule.conditions_match?(obj)
 93:                 matches = true
 94:                 
 95:                 # remove the rule from the working-set so it's not re-evaluated
 96:                 available_rules.delete(rule)
 97:                 
 98:                 # find all parameter-matching dependencies of this rule and
 99:                 # add them to the working-set.
100:                 if @dependencies.has_key?(rule.name)
101:                   available_rules += @dependencies[rule.name].select do |dependency|
102:                     dependency.parameters_match?(obj)
103:                   end
104:                 end
105:                 
106:                 # execute this rule
107:                 rule.call(obj)
108:                 
109:                 # break the current iteration and start back from the first rule defined.
110:                 break
111:               end # if rule.conditions_match?(obj)
112:               
113:             rescue RuleCheckError => e
114:               # log da error or sumpin
115:               available_rules.delete(e.rule)
116:             end # begin/rescue
117:             
118:           end # available_rules.each
119:           
120:         end while(matches && @assert)
121:         
122:       rescue RuleConsequenceError => rce
123:         # RuleConsequenceErrors are allowed to break out of the current assertion,
124:         # then the inner error is bubbled-up to the asserting code.
125:         raise rce.inner_error
126:       end
127:       
128:       @assert = false
129:       
130:       return status
131:     end
extend(name, &b)

Use in conjunction with Rools::RuleSet#with to create a Rools::Rule dependent on another. Dependencies are created through names (converted to strings and downcased), so lax naming can get you into trouble with creating dependencies or overwriting rules you didn’t mean to.

rule(name, &b)

rule creates a Rools::Rule. Make sure to use a descriptive name or symbol. For the purposes of extending Rules, all names are converted to strings and downcased.

Example

  rule 'ruby is the best' do
    condition { language.name.downcase == 'ruby' }
    consequence { "#{language.name} is the best!" }
  end

with(name, &b)

Used in conjunction with Rools::RuleSet#extend to create a dependent Rools::Rule

Example

  extend('ruby is the best').with('ruby rules the world') do
    condition { language.age > 15 }
    consequence { "In the year 2008 Ruby conquered the known universe" }
  end
